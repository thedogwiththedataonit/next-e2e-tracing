# =============================================================================
# next-app — environment variables
# =============================================================================
# Copy to `.env.local` for local dev, or configure each variable in the
# Next.js Vercel project's Environment Variables UI for preview/prod.
#
# The Flask API runs as a SEPARATE Vercel project — see the
# "flask-api project" section at the bottom for the env vars to configure
# on that project's side.

# --- Flask API connection -----------------------------------------------------
# Full URL (with scheme!) of the deployed Flask API, no trailing slash.
# The server-side /api/call route uses this to proxy requests; `@vercel/otel`
# automatically injects a W3C `traceparent` on this fetch, which the Flask
# side accepts via DD_TRACE_PROPAGATION_STYLE=tracecontext.
FLASK_API_URL=https://your-flask-api.vercel.app

# --- Datadog: server (Node.js) ------------------------------------------------
# Datadog API key — used by `lib/DDLogSubmission.ts` to ship structured
# server logs. Server-only — never expose as NEXT_PUBLIC_*.
DD_API_KEY=your-datadog-api-key-here

# Optional. Defaults to "datadoghq.com".
# DATADOG_SITE=datadoghq.com

# --- Datadog: browser RUM -----------------------------------------------------
# Consumed by `instrumentation-client.ts`, which calls `datadogRum.init()`
# with the official `@datadog/browser-rum-nextjs` plugin. All four required
# vars below must be set or RUM init is skipped (with a console warning).
#
# CRITICAL: NEXT_PUBLIC_* values are inlined into the client bundle at build
# time. Never put a server secret here — only the public RUM application id
# and client token belong here.
NEXT_PUBLIC_DATADOG_APPLICATION_ID=your-rum-application-id
NEXT_PUBLIC_DATADOG_CLIENT_TOKEN=your-rum-client-token
NEXT_PUBLIC_DATADOG_SERVICE_NAME=next-app
NEXT_PUBLIC_DATADOG_ENV=dev

# Optional. Defaults to "datadoghq.com".
# NEXT_PUBLIC_DATADOG_SITE=datadoghq.com

# Optional. Used to split RUM metrics by deploy. If unset, falls back to
# NEXT_PUBLIC_VERCEL_GIT_COMMIT_SHA (which Vercel populates on every build).
# NEXT_PUBLIC_DATADOG_VERSION=1.0.0

# Optional, comma-separated list of additional origins (or URL prefixes) that
# the browser RUM SDK should inject the W3C `traceparent` header onto. The
# Next.js same-origin (window.location.origin) is always included by default.
#
# Set this if you want browser code to ever call the Flask API directly and
# have those XHRs link into the Flask APM trace (the proxy-via-/api/call
# pattern in this repo does NOT need it).
# NEXT_PUBLIC_DATADOG_ALLOWED_TRACING_URLS=https://your-flask-api.vercel.app


# =============================================================================
# flask-api project — environment variables (FYI, set on the OTHER Vercel project)
# =============================================================================
# These are NOT consumed by next-app; they are the env vars you should
# configure on the flask-api Vercel project so that end-to-end tracing works.
#
# DD_API_KEY=<same Datadog API key>
# DD_SITE=datadoghq.com
# DD_ENV=prod                 # or `preview` / `dev`
# DD_SERVICE=flask-api
# DD_VERSION=$VERCEL_GIT_COMMIT_SHA
# DD_TRACE_ENABLED=true
# DD_TRACE_PROPAGATION_STYLE=tracecontext   # CRITICAL — accepts traceparent
# DD_TRACE_OTEL_ENABLED=true                # if Flask emits OTel spans
#
# CORS: only required if the BROWSER calls Flask directly. The default
# proxy-via-/api/call pattern in this repo does not need CORS.
# ALLOWED_ORIGIN=https://your-next-app.vercel.app
